 48        """
 49        Decorator that applies a column-wise statistical function to all numeric columns
 50        of `self.data` and returns a dictionary with keys named appropriately.
 51        """
 52

 60    def to_frame(func: Callable) -> Callable:
 61        """Decorator: Applies per-column expressions and evaluates with .with_columns(...)"""
 62
 63        @wraps(func)
 64        def wrapper(self, *args, **kwargs):
 65            return self.all.select(
 66                [pl.col(name) for name in self.data.date_col]
 67                + [func(self, pl.col(name), *args, **kwargs).alias(name) for name in self.data.assets]
 68            )

122    @to_frame
123    def drawdown(self, series: pl.Expr, compounded=False, initial=1.0) -> pl.Expr:
124        """
125        Computes drawdown from the high-water mark.
126
127        Args:
128            series (pl.Expr): Polars expression for the return series.
129            compounded (bool): Whether to use compounded returns.
130            initial (float): Initial portfolio value (default is 1).
131
132        Returns:
133            pl.Expr: A Polars expression representing the drawdown.
134        """
135        if compounded:
136            # First compute cumulative compounded returns
137            equity = initial * (1 + series).cum_prod()
138        else:
139            # Simple cumulative sum of returns
140            equity = initial + series.cum_sum()
141
142        # equity = self.price(series, compounded, initial=initial)
143        return -100 * ((equity / equity.cum_max()) - 1)

284    def kelly_criterion(self):
285        """
286        Calculates the optimal capital allocation (Kelly Criterion) per column:
287        f* = [(b * p) - q] / b
288        where:
289          - b = payoff ratio
290          - p = win rate
291          - q = 1 - p
292        """
293        b = self.payoff_ratio()
294        p = self.win_rate()
295
296        return {
297            col: ((b[col] * p[col]) - (1 - p[col])) / b[col]
298            # if b[col] not in (None, 0) and p[col] is not None else None
299            for col in b
300        }

334    @columnwise_stat
335    def sortino(self, series: pl.Series, periods=252):
336        """
337        Calculates the Sortino ratio: mean return divided by downside deviation.
338        Based on Red Rock Capital's Sortino ratio paper.
339        """
340
341        downside_deviation = np.sqrt(((series.filter(series < 0)) ** 2).mean())
342
343        ratio = series.mean() / downside_deviation
344        return ratio * np.sqrt(periods)

346    @to_frame
347    def rolling_sortino(self, series, rolling_period=126, periods_per_year=252):
348        mean_ret = series.rolling_mean(window_size=rolling_period)
349
350        # Rolling downside deviation (squared negative returns averaged over window)
351        downside = series.map_elements(lambda x: x**2 if x < 0 else 0.0).rolling_mean(window_size=rolling_period)
352
353        # Avoid division by zero
354        sortino = mean_ret / downside.sqrt().fill_nan(0).fill_null(0)
355        return sortino * (periods_per_year**0.5)

414    @columnwise_stat
415    def greeks(self, series, periods_per_year=252, benchmark=None):
416        """Calculates alpha and beta of the portfolio"""
417        # find covariance
418        benchmark_col = benchmark or self.data.benchmark.columns[0]
419
420        # Evaluate both series and benchmark as Series
421        df = self.all.select([series, pl.col(benchmark_col).alias("benchmark")])
422
423        # Drop nulls
424        df = df.drop_nulls()
425        matrix = df.to_numpy()
426
427        # Get actual Series
428        strategy_np = matrix[:, 0]
429        benchmark_np = matrix[:, 1]
430
431        # 2x2 covariance matrix: [[var_strategy, cov], [cov, var_benchmark]]
432        cov_matrix = np.cov(strategy_np, benchmark_np)
433
434        cov = cov_matrix[0, 1]
435        var_benchmark = cov_matrix[1, 1]
436
437        beta = cov / var_benchmark if var_benchmark != 0 else float("nan")
438        alpha = np.mean(strategy_np) - beta * np.mean(benchmark_np)
439
440        return {"alpha": alpha * periods_per_year, "beta": beta}

417        """Calculates alpha and beta of the portfolio"""
418        # find covariance
419        benchmark_col = benchmark or self.data.benchmark.columns[0]
420
421        # Evaluate both series and benchmark as Series
422        df = self.all.select([series, pl.col(benchmark_col).alias("benchmark")])

426        matrix = df.to_numpy()
427
428        # Get actual Series
429        strategy_np = matrix[:, 0]
430        benchmark_np = matrix[:, 1]
431
432        # 2x2 covariance matrix: [[var_strategy, cov], [cov, var_benchmark]]
433        cov_matrix = np.cov(strategy_np, benchmark_np)
434
435        cov = cov_matrix[0, 1]
436        var_benchmark = cov_matrix[1, 1]
437
438        beta = cov / var_benchmark if var_benchmark != 0 else float("nan")
439        alpha = np.mean(strategy_np) - beta * np.mean(benchmark_np)
440
441        return {"alpha": alpha * periods_per_year, "beta": beta}

Statistical analysis tools for financial returns data.

This class provides a comprehensive set of methods for calculating various financial metrics and statistics on returns data, including:

  • Basic statistics (mean, skew, kurtosis)
  • Risk metrics (volatility, value-at-risk, drawdown)
  • Performance ratios (Sharpe, Sortino, information ratio)
  • Win/loss metrics (win rate, profit factor, payoff ratio)
  • Rolling calculations (rolling volatility, rolling Sharpe)
  • Factor analysis (alpha, beta, R-squared)

The class is designed to work with the _Data class and operates on Polars DataFrames for efficient computation.

Attributes: data: The _Data object containing returns and benchmark data. all: A DataFrame combining all data (index, returns, benchmark) for easy access.
